388  void add_variable
389  (
390  const IndexType &v,
391  const FloatType &bias
392  )
393  {
394  FloatType b = bias;
395 
396  Vartype vartype = this->get_vartype();
397 
398  // handle the case that a different vartype is provided
399  if((vartype!=Vartype::NONE)&&(vartype!=m_vartype))
400  {
401  if((m_vartype == Vartype::SPIN)&&(vartype == Vartype::BINARY))
402  {
403  b /= 2;
404  m_offset += b;
405  }
406  else if((m_vartype == Vartype::BINARY)&&(vartype == Vartype::SPIN))
407  {
408  m_offset -= b;
409  b *= 2;
410  }
411  else
412  {
413  throw std::runtime_error("Unknown vartype");
414  }
415  }
416 
417  // Insert or assign the bias
418  FloatType value = 0;
419  if(m_linear.count(v) != 0)
420  {
421  value = m_linear[v];
422  }
423  insert_or_assign(m_linear, v, value + b);
424  }

449  void add_interaction
450  (
451  const IndexType &arg_u,
452  const IndexType &arg_v,
453  const FloatType &bias
454  )
455  {
456  IndexType u = std::min(arg_u, arg_v);
457  IndexType v = std::max(arg_u, arg_v);
458 
459  Vartype vartype = this->get_vartype();
460 
461  if(u == v)
462  {
463  throw std::runtime_error("No self-loops allowed");
464  }
465  else
466  {
467  //Check vartype when m_linear.empty() is true
468  if (m_linear.empty() && m_vartype == Vartype::NONE) {
469  if (vartype != Vartype::NONE) {
470  m_vartype = vartype;
471  }
472  else {
473  throw std::runtime_error("Binary quadratic model is empty. Please set vartype to Vartype::SPIN or Vartype::BINARY");
474  }
475  }
476 
477  FloatType b = bias;
478  if((vartype!=Vartype::NONE)&&(vartype!=m_vartype))
479  {
480  if((m_vartype == Vartype::SPIN)&&(vartype == Vartype::BINARY))
481  {
482  //convert from binary to spin
483  b /= 4;
484  add_offset(b);
485  add_variable(u, b);
486  add_variable(v, b);
487  }
488  else if((m_vartype == Vartype::BINARY)&&(vartype == Vartype::SPIN))
489  {
490  //convert from spin to binary
491  add_offset(b);
492  add_variable(u, -2 * b);
493  add_variable(v, -2 * b);
494  b *= 4;
495  }
496  else
497  {
498  throw std::runtime_error("Unknown vartype");
499  }
500  }
501  else
502  {
503  if(m_linear.count(u) == 0)
504  {
505  add_variable(u, 0);
506  }
507  if(m_linear.count(v) == 0)
508  {
509  add_variable(v, 0);
510  }
511  }
512 
513  FloatType value = 0;
514  std::pair<IndexType, IndexType> p1 = std::make_pair(u, v);
515  if(m_quadratic.count(p1) != 0)
516  {
517  value = m_quadratic[p1];
518  }
519  insert_or_assign(m_quadratic, p1, value + b);
520  }
521  }

585  void remove_interaction
586  (
587  const IndexType &arg_u,
588  const IndexType &arg_v
589  )
590  {
591  IndexType u = std::min(arg_u, arg_v);
592  IndexType v = std::max(arg_u, arg_v);
593  auto p = std::make_pair(u, v);
594  if(m_quadratic.count(p)!=0)
595  {
596  m_quadratic.erase(p);
597  }
598 
599  bool u_flag = true;
600  for(auto &it : m_quadratic)
601  {
602  if((it.first.first == u) || (it.first.second == u)){
603  u_flag = false;
604  break;
605  }
606  }
607 
608  if(u_flag && (m_linear[u] == 0)){
609  remove_variable(u);
610  }
611 
612  bool v_flag = true;
613  for(auto &it : m_quadratic)
614  {
615  if((it.first.first == v) || (it.first.second == v)){
616  v_flag = false;
617  break;
618  }
619  }
620 
621  if(v_flag && (m_linear[v] == 0)){
622  remove_variable(v);
623  }
624  }

671  void scale
672  (
673  const FloatType &scalar,
674  const std::vector<IndexType> &ignored_variables = {},
675  const std::vector<std::pair<IndexType, IndexType>> &ignored_interactions = {},
676  const bool ignored_offset = false
677  )
678  {
679  // scaling linear
680  for(auto &it : m_linear)
681  {
682  if(std::find(ignored_variables.begin(), ignored_variables.end(), it.first) != ignored_variables.end() || ignored_variables.empty())
683  {
684  it.second *= scalar;
685  }
686  }
687 
688  // scaling quadratic
689  for(auto &it : m_quadratic)
690  {
691  if(std::find(ignored_interactions.begin(), ignored_interactions.end(), it.first) != ignored_interactions.end() || ignored_variables.empty())
692  {
693  it.second *= scalar;
694  }
695  }
696 
697  // scaling offset
698  if(!ignored_offset)
699  {
700  m_offset *= scalar;
701  }
702  }

715  void normalize
716  (
717  const std::pair<FloatType, FloatType> &bias_range = {1.0, 1.0},
718  const bool use_quadratic_range = false,
719  const std::pair<FloatType, FloatType> &quadratic_range = {1.0, 1.0},
720  const std::vector<IndexType> &ignored_variables = {},
721  const std::vector<std::pair<IndexType, IndexType>> &ignored_interactions = {},
722  const bool ignored_offset = false
723  )
724  {
725  if (m_linear.empty()) {
726  return;
727  }
728  // parse range
729  std::pair<FloatType, FloatType> l_range = bias_range;
730  std::pair<FloatType, FloatType> q_range;
731  if(!use_quadratic_range)
732  {
733  q_range = bias_range;
734  }
735  else
736  {
737  q_range = quadratic_range;
738  }
739 
740  // calculate scaling value
741  auto comp = [](const auto &a, const auto &b) { return a.second < b.second; };
742  auto it_lin_min = std::min_element(m_linear.begin(), m_linear.end(), comp);
743  auto it_lin_max = std::max_element(m_linear.begin(), m_linear.end(), comp);
744  auto it_quad_min = std::min_element(m_quadratic.begin(), m_quadratic.end(), comp);
745  auto it_quad_max = std::max_element(m_quadratic.begin(), m_quadratic.end(), comp);
746 
747  std::vector<FloatType> v_scale =
748  {
749  it_lin_min->second / l_range.first,
750  it_lin_max->second / l_range.second,
751  it_quad_min->second / q_range.first,
752  it_quad_max->second / q_range.second
753  };
754  FloatType inv_scale = *std::max_element(v_scale.begin(), v_scale.end());
755 
756  // scaling
757  if(inv_scale != 0.0)
758  {
759  scale(1.0 / inv_scale, ignored_variables, ignored_interactions, ignored_offset);
760  }
761  }

770  (
771  const IndexType &v,
772  const int32_t &value
773  )
774  {

786  interactions.push_back(it.first);

788  }

815  void flip_variable
816  (
817  const IndexType &v
818  )
819  {
820  // check variable
821  if(m_linear.count(v)==0)
822  {
823  throw std::runtime_error("not a variable in the binary quadratic model.");
824  }
825 
826  if(m_vartype == Vartype::SPIN)
827  {
828  m_linear[v] *= -1.0;
829  for(auto &it : m_quadratic)
830  {
831  if(it.first.first == v || it.first.second == v)
832  {
833  it.second *= -1.0;
834  }
835  }
836  }
837  else if(m_vartype == Vartype::BINARY)
838  {
839  add_offset(m_linear[v]);
840  m_linear[v] *= -1.0;
841 
842  for(auto &it : m_quadratic)
843  {
844  if(it.first.first == v)
845  {
846  m_linear[it.first.second] += it.second;
847  it.second *= -1.0;
848  }
849  else if(it.first.second == v)
850  {
851  m_linear[it.first.first] += it.second;
852  it.second *= -1.0;
853  }
854  }
855  }

864  void contract_variables
865  (
866  const IndexType &u,
867  const IndexType &v
868  )
869  {
870  // check variable
871  if(m_linear.count(v)==0)
872  {
873  throw std::runtime_error("not a variable in the binary quadratic model.");
874  }
875  if(m_linear.count(u)==0)
876  {
877  throw std::runtime_error("not a variable in the binary quadratic model.");
878  }
879 
880  auto p1 = std::make_pair(u, v);
881  auto p2 = std::make_pair(v, u);
882  if(m_quadratic.count(p1) != 0)
883  {
884  if(m_vartype == Vartype::BINARY)
885  {
886  add_variable(u, m_quadratic[p1]);
887  }
888  else if(m_vartype == Vartype::SPIN)
889  {
890  add_offset(m_quadratic[p1]);
891  }
892  remove_interaction(u, v);
893  }
894  if(m_quadratic.count(p2) != 0)
895  {
896  if(m_vartype == Vartype::BINARY)
897  {
898  add_variable(u, m_quadratic[p2]);
899  }
900  else if(m_vartype == Vartype::SPIN)
901  {
902  add_offset(m_quadratic[p2]);
903  }
904  remove_interaction(v, u);
905  }
906 
907  std::vector<std::pair<IndexType, IndexType>> interactions;
908  for(auto &it : m_quadratic)
909  {
910  if(it.first.first == v)
911  {
912  add_interaction(u, it.first.second, it.second);
913  interactions.push_back(it.first);
914  }
915  else if(it.first.second == v)
916  {
917  add_interaction(it.first.first, u, it.second);
918  interactions.push_back(it.first);
919  }
920  }
921  remove_interactions_from(interactions);
922 
923  add_variable(u, m_linear[v]);
924  remove_variable(v);
925  }

935  void change_vartype
936  (
937  const Vartype &vartype
938  )
939  {
940  Linear<IndexType, FloatType> linear;
941  Quadratic<IndexType, FloatType> quadratic;
942  FloatType offset = 0.0;
943 
944  if(m_vartype == Vartype::BINARY && vartype == Vartype::SPIN) // binary -> spin
945  {
946  std::tie(linear, quadratic, offset) = binary_to_spin(m_linear, m_quadratic, m_offset);
947  }
948  else if(m_vartype == Vartype::SPIN && vartype == Vartype::BINARY) // spin -> binary
949  {
950  std::tie(linear, quadratic, offset) = spin_to_binary(m_linear, m_quadratic, m_offset);
951  }
952  else
953  {
954  std::tie(linear, quadratic, offset) = std::tie(m_linear, m_quadratic, m_offset);
955  }
956 
957  //inplace
958  m_linear = linear;
959  m_quadratic = quadratic;
960  m_offset = offset;
961  m_vartype = vartype;
962  }

972  (

1008  /* Static methods */
1018  static std::tuple<Linear<IndexType, FloatType>, Quadratic<IndexType, FloatType>, FloatType> spin_to_binary
1019  (
1020  const Linear<IndexType, FloatType> &linear,
1021  const Quadratic<IndexType, FloatType> &quadratic,
1022  const FloatType &offset
1023  )
1024  {
1025  Linear<IndexType, FloatType> new_linear;
1026  Quadratic<IndexType, FloatType> new_quadratic;
1027  FloatType new_offset = offset;
1028  FloatType linear_offset = 0.0;
1029  FloatType quadratic_offset = 0.0;
1030 
1031  for(auto &it : linear)
1032  {
1033  insert_or_assign(new_linear, it.first, static_cast<FloatType>(2.0 * it.second));
1034  linear_offset += it.second;
1035  }
1036 
1037  for(auto &it : quadratic)
1038  {
1039  insert_or_assign(new_quadratic, it.first, static_cast<FloatType>(4.0 * it.second));
1040  new_linear[it.first.first] -= 2.0 * it.second;
1041  new_linear[it.first.second] -= 2.0 * it.second;
1042  quadratic_offset += it.second;
1043  }
1044 
1045  new_offset += quadratic_offset - linear_offset;
1046 
1047  return std::make_tuple(new_linear, new_quadratic, new_offset);
