57 template <typename IndexType>
58 void FormatPolynomialKey(std::vector<IndexType> *key, const Vartype &vartype) {
59  if ((*key).size() <= 1) {
60  return;
61  }
62  else if ((*key).size() == 2) {
63  if ((*key)[0] == (*key)[1]) {
64  if (vartype == Vartype::SPIN) {
65  (*key).clear();
66  return;
67  }
68  else if (vartype == Vartype::BINARY) {
69  (*key).pop_back();
70  return;
71  }
72  else {
73  throw std::runtime_error("Unknown vartype detected");
74  }
75  }
76  else if ((*key)[0] < (*key)[1]) {
77  return;
78  }
79  else {
80  std::swap((*key)[0], (*key)[1]);
81  return;
82  }
83  }
84  else {
85  std::sort((*key).begin(), (*key).end());
86  if (vartype == Vartype::SPIN) {
87  for (int64_t i = static_cast<int64_t>((*key).size()) - 1; i > 0; --i) {
88  if ((*key)[i] == (*key)[i - 1]) {
89  std::swap((*key)[i], (*key).back());
90  (*key).pop_back();
91  --i;
92  std::swap((*key)[i], (*key).back());
93  (*key).pop_back();
94  }
95  }
96  return;
97  }
98  else if (vartype == Vartype::BINARY) {
99  (*key).erase(std::unique((*key).begin(), (*key).end()), (*key).end());
100  return;
101  }
102  else {
103  throw std::runtime_error("Unknown vartype detected");
104  }
105  }
106 }
