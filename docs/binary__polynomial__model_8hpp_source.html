217  BinaryPolynomialModel(
218  const std::vector<IndexType> &variables,
219  const PolynomialKeyList<std::size_t> &poly_key_distance_list,
220  const PolynomialValueList<FloatType> &poly_value_list,
221  const Vartype vartype ) :
222  vartype_( vartype ) {
223  if ( vartype_ == Vartype::NONE ) {
224  throw std::runtime_error( "Unknown vartype detected" );
225  }
226 
227  if ( poly_key_distance_list.size() != poly_value_list.size() ) {
228  throw std::runtime_error( "The sizes of key_list and value_list must match each other" );
229  }
230 
231  variables_ = std::unordered_set<IndexType>( variables.begin(), variables.end() );
232 
233  if ( variables_.size() != variables.size() ) {
234  throw std::runtime_error( "Unknown error. It seems that the input variables contain the same variables" );
235  }
236 
237  std::size_t num_interactions = poly_key_distance_list.size();
238  poly_key_list_.resize( num_interactions );
239  poly_value_list_.resize( num_interactions );
240 
241 #pragma omp parallel for
242  for ( int64_t i = 0; i < ( int64_t )num_interactions; ++i ) {
243  std::vector<IndexType> temp;
244  for ( const auto &it : poly_key_distance_list[ i ] ) {
245  temp.push_back( variables[ it ] );
246  }
247  std::sort( temp.begin(), temp.end() );
248  poly_key_list_[ i ] = temp;
249  poly_value_list_[ i ] = poly_value_list[ i ];
250  }
251 
252  for ( std::size_t i = 0; i < num_interactions; ++i ) {
253  poly_key_inv_[ poly_key_list_[ i ] ] = i;
254  for ( const auto &it : poly_key_list_[ i ] ) {
255  each_variable_num_[ it ]++;
256  }
257  }
258 
259  UpdateVariablesToIntegers();
260  }

447  void RemoveInteraction( std::vector<IndexType> &key ) {
448  FormatPolynomialKey( &key, vartype_ );
449  if ( poly_key_inv_.count( key ) == 0 ) {
450  return;
451  }
452 
453  for ( const auto &index : key ) {
454  if ( each_variable_num_[ index ] >= 2 ) {
455  each_variable_num_[ index ]--;
456  } else if ( each_variable_num_[ index ] == 1 ) {
457  each_variable_num_.erase( index );
458  variables_.erase( index );
459  relabel_flag_for_variables_to_integers_ = true;
460  }
461  }
462 
463  std::size_t inv = poly_key_inv_[ key ];
464 
465  std::swap( poly_key_inv_[ key ], poly_key_inv_[ poly_key_list_.back() ] );
466  poly_key_inv_.erase( key );
467 
468  std::swap( poly_key_list_[ inv ], poly_key_list_.back() );
469  poly_key_list_.pop_back();
470 
471  std::swap( poly_value_list_[ inv ], poly_value_list_.back() );
472  poly_value_list_.pop_back();
473  }

525  void AddInteraction( std::vector<IndexType> &key, const FloatType &value, const Vartype vartype = Vartype::NONE ) {
526  if ( std::abs( value ) <= 0.0 ) {
527  return;
528  }
529 
530  if ( vartype_ == vartype || vartype == Vartype::NONE ) {
531  FormatPolynomialKey( &key, vartype_ );
532  SetKeyAndValue( key, value );
533  } else {
534  const std::size_t original_key_size = key.size();
535  const std::size_t changed_key_list_size = IntegerPower( 2, original_key_size );
536 
537  if ( vartype_ == Vartype::SPIN && vartype == Vartype::BINARY ) {
538  FormatPolynomialKey( &key, vartype );
539  for ( std::size_t i = 0; i < changed_key_list_size; ++i ) {
540  const auto changed_key = GenerateChangedKey( key, i );
541  int sign = ( ( original_key_size - changed_key.size() ) % 2 == 0 ) ? 1.0 : -1.0;
542  SetKeyAndValue( changed_key, value * IntegerPower( 2, changed_key.size() ) * sign );
543  }
544  } else if ( vartype_ == Vartype::BINARY && vartype == Vartype::SPIN ) {
545  FormatPolynomialKey( &key, vartype );
546  FloatType changed_value = value * ( 1.0 / changed_key_list_size );
547  for ( std::size_t i = 0; i < changed_key_list_size; ++i ) {
548  SetKeyAndValue( GenerateChangedKey( key, i ), changed_value );
549  }
550  } else {
551  throw std::runtime_error( "Unknown vartype error" );
552  }
553  }
554  }

618  FloatType Energy( const Sample<IndexType> &sample, bool omp_flag = true ) const {
619  if ( sample.size() != GetNumVariables() ) {
620  throw std::runtime_error( "The size of sample must be equal to num_variables" );
621  }
622 
623  if ( GetNumInteractions() == 0 ) {
624  return 0.0;
625  }
626 
627  std::size_t num_interactions = GetNumInteractions();
628  FloatType val = 0.0;
629 
630  if ( omp_flag ) {
631 #pragma omp parallel for reduction( + : val )
632  for ( int64_t i = 0; i < ( int64_t )num_interactions; ++i ) {
633  int32_t spin_multiple = 1;
634  for ( const auto &index : poly_key_list_[ i ] ) {
635  spin_multiple *= sample.at( index );
636  if ( spin_multiple == 0.0 ) {
637  break;
638  }
639  }
640  val += spin_multiple * poly_value_list_[ i ];
641  }
642  } else {
643  for ( std::size_t i = 0; i < num_interactions; ++i ) {
644  int32_t spin_multiple = 1;
645  for ( const auto &index : poly_key_list_[ i ] ) {
646  spin_multiple *= sample.at( index );
647  if ( spin_multiple == 0.0 ) {
648  break;
649  }
650  }
651  val += spin_multiple * poly_value_list_[ i ];
652  }
653  }
654  return val;
655  }

733  void Scale(
734  const FloatType scalar,
735  const PolynomialKeyList<IndexType> &ignored_interactions = {},
736  const bool ignored_offset = false ) {
737 
738  std::size_t num_interactions = GetNumInteractions();
739 
740  for ( std::size_t i = 0; i < num_interactions; ++i ) {
741  poly_value_list_[ i ] *= scalar;
742  }
743 
744  FloatType scalar_inv = 1.0 / scalar;
745  for ( const auto &key : ignored_interactions ) {
746  if ( poly_key_inv_.count( key ) != 0 ) {
747  poly_value_list_[ poly_key_inv_[ key ] ] *= scalar_inv;
748  }
749  }
750 
751  if ( ignored_offset == true && poly_key_inv_.count( std::vector<IndexType>{} ) != 0
752  && std::count( ignored_interactions.begin(), ignored_interactions.end(), std::vector<IndexType>{} ) == 0 ) {
753  poly_value_list_[ poly_key_inv_[ std::vector<IndexType>{} ] ] *= scalar_inv;
754  }
755  }

762  void normalize(
763  const std::pair<FloatType, FloatType> &range = { 1.0, 1.0 },
764  const PolynomialKeyList<IndexType> &ignored_interactions = {},
765  const bool ignored_offset = false ) {
766 
767  if ( GetNumInteractions() == 0 ) {
768  return;
769  }
770 
771  FloatType max_poly_value = poly_value_list_[ 0 ];
772  FloatType min_poly_value = poly_value_list_[ 0 ];
773 
774  for ( const auto &poly_value : poly_value_list_ ) {
775  if ( max_poly_value < poly_value ) {
776  max_poly_value = poly_value;
777  }
778  if ( min_poly_value > poly_value ) {
779  min_poly_value = poly_value;
780  }
781  }
782 
783  FloatType inv_scale = std::max( min_poly_value / range.first, max_poly_value / range.second );
784 
785  if ( inv_scale != 0.0 ) {
786  Scale( 1.0 / inv_scale, ignored_interactions, ignored_offset );
787  }
788  }

839  Polynomial<IndexType, FloatType> ToHubo() const {
840  if ( vartype_ == Vartype::BINARY ) {
841  return GetPolynomial();
842  }
843  Polynomial<IndexType, FloatType> poly_map;
844  std::size_t num_interactions = GetNumInteractions();
845  for ( std::size_t i = 0; i < num_interactions; ++i ) {
846  const std::vector<IndexType> &original_key = poly_key_list_[ i ];
847  const FloatType original_value = poly_value_list_[ i ];
848  const std::size_t original_key_size = original_key.size();
849  const std::size_t changed_key_list_size = IntegerPower( 2, original_key_size );
850 
851  for ( std::size_t j = 0; j < changed_key_list_size; ++j ) {
852  const auto changed_key = GenerateChangedKey( original_key, j );
853  int sign = ( ( original_key_size - changed_key.size() ) % 2 == 0 ) ? 1.0 : -1.0;
854  FloatType changed_value = original_value * IntegerPower( 2, changed_key.size() ) * sign;
855  poly_map[ changed_key ] += changed_value;
